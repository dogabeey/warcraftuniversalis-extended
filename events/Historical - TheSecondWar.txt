namespace = the_second_war

#Form Horde
country_event = {
	id = the_second_war.1
	title = horde_forming.title
	desc = {
		trigger = { has_global_flag = tfw_orc_victory_path }
		desc = horde_forming.desc.1
	}
	desc = {
		trigger = { has_global_flag = tfw_human_victory_path }
		desc = horde_forming.desc.2
	}
	picture = ORCISH_INVASION_eventPicture

    hidden = no
	fire_only_once = yes
	major = yes
    
	trigger = {
        has_global_flag = the_second_war
	}
	
    mean_time_to_happen = {
		days = 1
    }
    
    immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					culture_group = culture_group_orc
					NOT = {
						any_country = {
							total_development = PREV
						}
					}
				}
				save_event_target_as = most_developed_horde
			}
			
			event_target:most_developed = {
				add_country_modifier = {
					name = horde_head
					duration = -1
				}
				add_country_modifier = {
					name = horde_member
					duration = -1
				}
			}
		}
    }
    
	option = {
		name = horde_forming.opt1
    }
}
